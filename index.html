<button class="red" onclick="reset()">Reset</button>


<script>

const menu = [

  { name: "Chai", price: 30 },

  { name: "Coffee", price: 35 },

  { name: "Samosa", price: 30 },

  { name: "Kachori", price: 30 },

  { name: "Sandwich", price: 70 }

];


let total = 0;


function render() {

  const container = document.getElementById("menu");

  container.innerHTML = "";


  menu.forEach(item => {

    const div = document.createElement("div");

    div.className = "item";


    div.innerHTML = `

      <b>${item.name}</b> - ₹${item.price}

      <br/>

      <button class="green" onclick="add(${item.price})">Add</button>

    `;


    container.appendChild(div);

  });

}


function add(price) {

  total += price;

  document.getElementById("total").innerText = total;

}


function reset() {

  total = 0;

  document.getElementById("total").innerText = 0;

}


render();

</script>
